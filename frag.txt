!!ARBfp1.0
#DECLARATION
TEMP fCol;
MOV fCol, fragment.color;
#DECLARATION
TEMP fTex;
MOV fTex, fragment.texcoord;
#DECLARATION
TEMP eyeNorm;
#DECLARATION
TEMP coeff;
#DECLARATION
TEMP shade;
#DECLARATION
PARAM lVec = program.env[1];
#DECLARATION
PARAM lHalf = state.light[0].half;
#DECLARATION
TEMP tempVar0;
MOV tempVar0.x, 1.000000;
MOV tempVar0.y, 0.000000;
MOV tempVar0.z, 0.000000;
MOV tempVar0.w, 1.000000;
PARAM red = tempVar0;
MOV eyeNorm, fTex;
TEMP tempVar1;
DP3 tempVar1, eyeNorm, eyeNorm;
MOV eyeNorm.w, tempVar1;
TEMP tempVar2;
RSQ tempVar2, eyeNorm.w;
MOV eyeNorm.w, tempVar2;
TEMP tempVar3;
MUL tempVar3, eyeNorm, eyeNorm.w;
MOV eyeNorm, tempVar3;
TEMP tempVar4;
MUL tempVar4, state.lightmodel.ambient, fCol;
MOV shade, tempVar4;
TEMP tempVar5;
DP3 tempVar5, lVec, eyeNorm;
MOV coeff.x, tempVar5;
TEMP tempVar6;
DP3 tempVar6, lHalf, eyeNorm;
MOV coeff.y, tempVar6;
MOV coeff.w, state.material.shininess.x;
TEMP tempVar7;
LIT tempVar7, coeff;
MOV coeff, tempVar7;
TEMP tempVar8;
MUL tempVar8, coeff.y, fCol;
TEMP tempVar9;
ADD tempVar9, shade, tempVar8;
MOV shade, tempVar9;
TEMP tempVar10;
MUL tempVar10, coeff.z, fCol;
TEMP tempVar11;
ADD tempVar11, shade, tempVar10;
MOV shade, tempVar11;
MOV result.color, shade;
END
